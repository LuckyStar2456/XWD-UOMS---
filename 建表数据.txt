-- 用户管理表(sys_user)
CREATE TABLE sys_user (
  id BIGINT AUTO_INCREMENT PRIMARY KEY, 
  username VARCHAR(50) NOT NULL COMMENT '账号：学号 / 学工号 等',
  password VARCHAR(100) NOT NULL COMMENT '密码',
  name VARCHAR(50) NOT NULL COMMENT '姓名',
  grade VARCHAR(20) NOT NULL COMMENT '年级',
  phone VARCHAR(20) NOT NULL COMMENT '联系电话',
  email VARCHAR(100) COMMENT '邮箱',
  org_id BIGINT COMMENT '所属组织 ID', 
  dept_id BIGINT COMMENT '所属部门 ID', 
  role_id BIGINT NOT NULL COMMENT '权限角色ID：1 超级管理员 / 2 部门部员 / 3 部门部长 / 4 组织主席 ', 
  status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0 禁用 / 1 正常',
  is_deleted TINYINT NOT NULL DEFAULT 0 COMMENT '状态：0 正常 / 1 注销',
  create_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  update_date DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

  INDEX idx_sys_user_org_id (org_id),
  INDEX idx_sys_user_dept_id (dept_id),
  INDEX idx_sys_user_role_id (role_id),

  UNIQUE INDEX uk_sys_user_username (username),
  UNIQUE INDEX uk_sys_user_phone (phone)
) ENGINE=InnoDB 
DEFAULT CHARACTER SET utf8mb4 
COLLATE utf8mb4_unicode_ci 
COMMENT '用户管理表';
